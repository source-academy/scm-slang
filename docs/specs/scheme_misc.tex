\subsection*{Core Library Summary}

\begin{itemize}
\item Errors: \texttt{error}
\item Booleans: \texttt{boolean?}, \texttt{not}, \texttt{and}, \texttt{or}
\item Numbers: \texttt{number?}, \texttt{integer?}, \texttt{rational?}, \texttt{real?},
  \texttt{complex?}, \texttt{exact?}, \texttt{inexact?}, \texttt{+}, \texttt{-},
  \texttt{*}, \texttt{/}, \texttt{=}, \texttt{<}, \texttt{<=}, \texttt{>}, \texttt{>=},
  \texttt{abs}, \texttt{quotient}, \texttt{remainder}, \texttt{modulo}, \texttt{gcd},
  \texttt{lcm}, \texttt{exp}, \texttt{log}, \texttt{sqrt}, \texttt{sin}, \texttt{cos},
  \texttt{tan}, \texttt{floor}, \texttt{ceiling}, \texttt{round}
\item Pairs and lists: \texttt{cons}, \texttt{car}, \texttt{cdr}, \texttt{set-car!},
  \texttt{set-cdr!}, \texttt{list}, \texttt{null?}, \texttt{list?}, \texttt{append},
  \texttt{map}, \texttt{fold}, \texttt{filter}, \texttt{length}
\item Vectors: \texttt{vector}, \texttt{vector?}, \texttt{vector-length},
  \texttt{vector-ref}, \texttt{vector-set!}, \texttt{vector->list}, \texttt{list->vector}
\item Symbols: \texttt{symbol?}, \texttt{symbol=?}, \texttt{string->symbol}
\item Strings: \texttt{string}, \texttt{string-length}, \texttt{string-ref},
  \texttt{substring}, \texttt{string-append}, \texttt{string->number}, \texttt{number->string}
\item Equality: \texttt{eq?}, \texttt{eqv?}, \texttt{equal?}
\item Promises: \texttt{make-promise}, \texttt{promise?}, \texttt{force},
  \texttt{promise-forced?}
\item Output: \texttt{display} (returns its argument)
\end{itemize}
